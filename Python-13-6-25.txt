#Sum of Digits:
 Given a number, find the sum of its digits (e.g., 123 â 6).
 
 def sum_of_digit(num):
    sum=0
    for i in num:
        sum+=i
    return(sum)
 num = list(map(int, input("Enter: ").split()))
 print(sum_of_digit(num))

#Count Vowels in a String:
 Input a string and count how many vowels it contains.
 
 def vowel_count(s):
    count=0
    vowels=["a","e","i","o","u"]
    for i in s.lower():
        if i in vowels:
            count+=1
    return(count)
 s=input()
 print(vowel_count(s))

#Find Maximum in a List:
 Given a list of numbers, find and print the maximum number.

def max_num(num):
    max=0
    while num>0:
        digit= num%10 
        if digit>max:
            max=digit
        num//=10
    return(max)
num=int(input())
print(max_num(num))
